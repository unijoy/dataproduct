Prediction

The prediction function

mypredciton <- function(cyl,hp,wt){
    if(cyl != 0 & hp != 0 & wt !=0 ){
        modelFit <- lm(mpg ~ cyl + hp + wt,data=mtcars)
        df<- data.frame(cyl=cyl,hp=hp,wt=wt)
        predict(modelFit,newdata = df)    
    }

}
